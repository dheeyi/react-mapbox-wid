## Install

```bash
npm install --save react-mapbox-wid
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-mapbox-wid'
import 'react-mapbox-wid/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
